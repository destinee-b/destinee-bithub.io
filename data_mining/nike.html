import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

#import
temp_df = pd.read_csv("nike_sales_2024.csv")

region_sales = temp_df.groupby("Region")["Revenue_USD"].sum().reset_index()
region_sales = region_sales.sort_values(by="Revenue_USD", ascending=False)
month_order = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"]
temp_df["Month"] = pd.Categorical(temp_df["Month"], categories=month_order, ordered=True)

#graph for sales by region
plt.figure(figsize=(10, 5))
sns.barplot(x="Revenue_USD", y="Region", data=region_sales)
plt.title("Total Nike Revenue by Region (2024)", fontsize=14, fontweight="bold")
plt.xlabel("Total Revenue (USD)")
plt.ylabel("Region")
plt.grid(axis="x", linestyle="--", alpha=0.7)

plt.show()

# graph for monthly sales
monthly_sales = temp_df.groupby("Month", observed=False)["Revenue_USD"].sum().reset_index()
monthly_sales = monthly_sales.sort_values(by="Month", key=lambda x: x.map(lambda m: month_order.index(m)))
plt.figure(figsize=(10, 5))
sns.lineplot(x="Month", y="Revenue_USD", data=monthly_sales, marker='o')
plt.title("Nike Monthly Sales Trend (2024)", fontsize=14, fontweight="bold")
plt.xlabel("Month")
plt.ylabel("Total Revenue (USD)")
plt.xticks(rotation=45)
plt.grid(axis="y", linestyle="--", alpha=0.7)
plt.show()

# graph for most popular products 
product_line_sales = temp_df.groupby("Product_Line")["Revenue_USD"].sum().reset_index().sort_values(by="Revenue_USD", ascending=False)
plt.figure(figsize=(10, 5))
sns.barplot(x="Revenue_USD", y="Product_Line", data=product_line_sales)
plt.title("Most Popular Nike Product Lines (2024)", fontsize=14, fontweight="bold")
plt.xlabel("Total Revenue (USD)")
plt.ylabel("Product Line")
plt.grid(axis="x", linestyle="--", alpha=0.7)
plt.show()
